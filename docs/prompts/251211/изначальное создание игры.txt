мне нужно разработать игру на Java без использования библиотек (на чистом Java Swing GUI)

Фичи:
- сделаем шутер со стрельбой в направлении мышки
- когда зажата левая кнопка мыши - стрельба
- бегать можно по WASD
- со всех сторон за пределами экрана появляются новые враги
- враги бегут на игрока напролом
- при убийстве врага из него иногда выпадают бонусы
- бонусы бывают: аптечка, увеличение отряда игрока (изначально мы - один кружок-солдат, с каждым бонусом, наш отряд увеличивается на еще один кружок-солдат, который тоже стреляет)
- враг при уничтожении взрывается (из него в случайных направлениях разлетаются осколки - эффект системы частиц)
- при взрыве врага происходит тряска камеры - за это ответственен такой-то класс

у меня уже есть MyFrame:

import javax.swing.*;
import java.awt.*;
import java.awt.image.BufferStrategy;

public class MyFrame extends JFrame {
    public MyFrame() {
        setSize(640, 480);
        setDefaultCloseOperation(EXIT_ON_CLOSE);

        setVisible(true); // делаем окно видимым только после того как оно полностью готово
    }

    @Override
    public void paint(Graphics g) {
        BufferStrategy bufferStrategy = getBufferStrategy(); // Обращаемся к стратегии буферизации
        if (bufferStrategy == null) { // Если она еще не создана
            createBufferStrategy(2); // то создаем ее
            bufferStrategy = getBufferStrategy(); // и опять обращаемся к уже наверняка созданной стратегии
        }
        g = bufferStrategy.getDrawGraphics(); // Достаем текущую графику (текущий буфер) - это наш холст для рисования (спрятанный от глаз пользователя)
        g.clearRect(0, 0, getWidth(), getHeight()); // Очищаем наш холст (ведь там остался предыдущий кадр)

        g.dispose();                // Освободить все временные ресурсы графики (после этого в нее уже нельзя рисовать)
        bufferStrategy.show();      // Сказать буферизирующей стратегии отрисовать новый буфер (т.е. поменять показываемый и обновляемый буферы местами)
    }
}

Main:

public class Main {
    public static void main(String[] args) throws InterruptedException {
        MyFrame frame = new MyFrame();
        while (true) {
            frame.repaint();

            // давайте отрисовывать окно не чаще чем раз в 10 миллисекунд - т.е. не чаще чем 100 раз в секунду
            Thread.sleep(10); // для этого ждем 10 миллисекунд прежде чем вновь вызвать frame.repaint();
            // это полезно для того чтобы не грузить процессор компьютера слишком сильно
        }
    }
}
